import type { MetadataRoute } from 'next';

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      {
        userAgent: '*',
        allow: '/',
        disallow: [
          '/admin',
        ],
      },
    ],
    host: 'https://www.mmsweden.se',
    sitemap: 'https://www.mmsweden.se/sitemap.xml',
  };
}
